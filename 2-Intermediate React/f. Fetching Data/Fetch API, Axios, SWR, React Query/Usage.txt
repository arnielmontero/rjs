Here are examples of fetching data in React using different tools:

1. Fetch API

import React, { useEffect, useState } from 'react';

function FetchExample() {
  const [data, setData] = useState(null);

  useEffect(() => {
    fetch('https://jsonplaceholder.typicode.com/posts/1')
      .then(res => res.json())
      .then(setData);
  }, []);

  if (!data) return <div>Loading...</div>;
  return <div>{data.title}</div>;
}

2. Axios

import React, { useEffect, useState } from 'react';
import axios from 'axios';

function AxiosExample() {
  const [data, setData] = useState(null);

  useEffect(() => {
    axios.get('https://jsonplaceholder.typicode.com/posts/1')
      .then(res => setData(res.data));
  }, []);

  if (!data) return <div>Loading...</div>;
  return <div>{data.title}</div>;
}

3. SWR
(Simple React Hooks for data fetching, caching, and revalidation)

import useSWR from 'swr';

function SWRExample() {
  const fetcher = url => fetch(url).then(res => res.json());
  const { data, error } = useSWR('https://jsonplaceholder.typicode.com/posts/1', fetcher);

  if (error) return <div>Error!</div>;
  if (!data) return <div>Loading...</div>;
  return <div>{data.title}</div>;
}

4. React Query

import { useQuery } from '@tanstack/react-query';

function ReactQueryExample() {
  const { data, isLoading, error } = useQuery({
    queryKey: ['post'],
    queryFn: () =>
      fetch('https://jsonplaceholder.typicode.com/posts/1').then(res => res.json()),
  });

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error!</div>;
  return <div>{data.title}</div>;
}
